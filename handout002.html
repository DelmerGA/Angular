<script>
function TodoCtrl($scope){
	<!-- Note: we are no longer calling
		our todo collection "firstTodos"-->
	$scope.todos = [{text: 'Practice Controllers',
	                 done:false},
	                {text: 'Play', done:false}
					]
	<!-- adding hasn't changed-->
	$scope.addTodo = function(){
		$scope.todos.push({text:$scope.formTodoText, done:false});
		$scope.formTodoText = '';	
	}
	<!--Total just returns length of todo array-->
	$scope.getTotalTodos = function(){
		return $scope.todos.length;
	}	
	<!--remove completed is open to interpretation-->
	$scope.removeCompleted = function(){
		for (var i=0; i < $scope.todos.length; i++) {
			if($scope.todos[i].done){
				$scope.todos.splice(i,1);
			}
		};
	}
}
</script>

<div ng-controller="TodoCtrl">
...
	<form class="form-horizontal">
		...
	<button class="btn" ng-click="clearCompleted()">Clear Complete</button>
	</form>
</div>
